using System.Diagnostics;
using QuestPDF.Fluent;
using QuestPDF.Helpers;
using QuestPDF.Infrastructure;

QuestPDF.Settings.License = LicenseType.Community;

bool isLetter = args.Length > 0 && args[0] == "1";
var pageSize = isLetter ? PageSizes.Letter : PageSizes.A4;
string outputFileName = $"lowfat_list_{(isLetter ? "letter" : "a4")}.pdf";

var headerColor = "#1B5E20";
var subheaderColor = "#388E3C";
var h2BgColor = "#E8F5E9";
var h2TextColor = "#2E7D32";

var sw = Stopwatch.StartNew();

Document.Create(container =>
{
    container.Page(page =>
    {
        page.Size(pageSize);
        page.Margin(36);
        page.DefaultTextStyle(x => x.FontSize(9).FontFamily("Ubuntu Sans"));

        // Header (H1)
        page.Header().Debug().Height(50).AlignCenter().Text(text =>
        {
            text.Span("LOW FAT ").Style(TextStyle.Default.FontSize(30).ExtraBold().FontColor(headerColor));
            text.Span("GROCERY LIST").Style(TextStyle.Default.FontSize(30).Bold().FontColor("#43A047"));
        });

        page.Content().Debug().PaddingTop(10).Table(table =>
        {
            table.ColumnsDefinition(columns =>
            {
                columns.RelativeColumn();
                columns.RelativeColumn();
                columns.RelativeColumn();
            });

            table.Cell().Debug().PaddingRight(6).ExtendVertical().Column(c =>
            {
                c.Spacing(10);
                c.Item().Debug().Column(s => {
                    CreateH2(s.Item(), "Fruits & Vegetables");
                    CreateSubheader(s.Item(), "Fresh Fruits");
                    CreateItem(s.Item(), "Apples");
                    CreateItem(s.Item(), "Bananas");
                    CreateItem(s.Item(), "Berries (strawberries, blueberries, raspberries)");
                });

                c.Item().Debug().Column(s => {
                    CreateH2(s.Item(), "Condiments & Dressings");
                    CreateItem(s.Item(), "Mustard");
                    CreateItem(s.Item(), "Salsa");
                });
            });

            table.Cell().Debug().PaddingHorizontal(6).Column(c =>
            {
                c.Spacing(10);
                c.Item().Debug().Column(s => {
                    CreateH2(s.Item(), "Grains & Legumes");
                    CreateSubheader(s.Item(), "Whole Grains");
                    CreateItem(s.Item(), "Brown Rice");
                });
            });

            table.Cell().Debug().PaddingLeft(6).Column(c =>
            {
                c.Spacing(10);
                c.Item().Debug().Column(s => {
                    CreateH2(s.Item(), "Snacks");
                    CreateItem(s.Item(), "Air-popped popcorn");
                });
            });
        });
    });
})
.GeneratePdf(outputFileName);

sw.Stop();

Console.WriteLine($"{outputFileName} created!");
Console.WriteLine($"Generation time: {sw.Elapsed.TotalSeconds:F3} seconds");

void CreateH2(IContainer container, string title)
{
    container.Background(h2BgColor)
             .PaddingVertical(2)
             .AlignCenter()
             .Text(title.ToUpper())
             .Style(TextStyle.Default.FontSize(11).SemiBold().FontColor(h2TextColor));
}

void CreateSubheader(IContainer container, string title)
{
    container.Debug().PaddingTop(5).Column(col => {
        col.Item().AlignCenter().Text(title.ToUpper()).Style(TextStyle.Default.Bold().FontColor(subheaderColor));
        col.Item().PaddingVertical(2).Height(1).Background(subheaderColor);
    });
}

void CreateItem(IContainer container, string text)
{
    container.Debug().PaddingBottom(2).Row(row =>
    {
        row.ConstantItem(15).PaddingTop(2).AlignCenter().Width(8).Height(8).Border(0.5f).BorderColor(Colors.Grey.Medium);
        
        row.RelativeItem().PaddingLeft(5).Text(text).FontSize(9);
    });
}

public static class DebugExtensions
{
    public static bool Enabled = true;

    public static IContainer Debug(this IContainer container)
    {
        if (!Enabled)
            return container;

        return container.Border(1).BorderColor(Colors.Red.Medium);
    }
}
